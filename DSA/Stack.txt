#include <stdio.h>
class stack {
private:
    LinkedList lst;
public:
    stack();
    void push(int x);
    int pop();
    void print();
    
    
};
stack::stack(){}

void stack::push(int x){
    lst.addFront(x);
    
}

int stack::pop(){
    int x = lst[0];
    return x;
}

void stack::print() {
    lst.print();
}

{
